jsx/tsx files

React embraces the fact that rendering logic is inherently coupled with other UI logic Code and UI are together in purpose 
                    
                        class Welcome extends React.Component {
                            render() {
                                return <h1>Hello, {this.props.name} </h1>;
                            }
                            componentWillMount(){
                                // called before component will be placed on DOM
                            }
                            componentDidMount(){
                                // caleed after component is placed on DOM
                            }
                        }

props and state

                        class Welcome extends React.Component {
